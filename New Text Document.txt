class ImageLoaderThread(QThread):

    image_loaded = pyqtSignal(dict)



    def __init__(self, paths):

        super().__init__()

        self.paths = paths



    def run(self):

        full_meta = []

        for p in self.paths:

            if os.path.exists(p):

                # ctime = creation time

                full_meta.append((p, os.path.getctime(p)))

        

        # Default sort: Oldest to Latest

        full_meta.sort(key=lambda x: x[1])

        self.image_loaded.emit({"type": "meta_ready", "data": full_meta})

        

        for idx, item in enumerate(full_meta):

            self.image_loaded.emit({"type": "add_icon", "item": item, "idx": idx})

            self.msleep(5)



class HoverIcon(QFrame):

    def __init__(self, img_path, meta_list, current_idx, parent_page, is_special=False):

        super().__init__()

        self.img_path = img_path

        self.meta_list = meta_list

        self.idx = current_idx

        self.parent_page = parent_page

        self.setFixedWidth(160)

        

        # Hover/Active Style

        self.normal_style = "QFrame { background: rgba(0,0,0,150); border: 2px solid white; border-radius: 12px; }"

        self.hover_style = "QFrame { background: rgba(0,242,255,40); border: 2px solid #00F2FF; border-radius: 12px; }"

        self.setStyleSheet(self.normal_style)



        layout = QVBoxLayout(self)

        layout.setContentsMargins(8, 8, 8, 8)



        self.display_label = QLabel()

        self.display_label.setFixedHeight(140)

        self.display_label.setAlignment(Qt.AlignmentFlag.AlignCenter)

        

        # File Icon or Image Logic

        ext = os.path.splitext(img_path)[1].lower()

        if ext in ('.jpg', '.jpeg', '.png', '.gif', '.bmp'):

            pix = QPixmap(img_path)

            if not pix.isNull():

                # Keep aspect ratio without stretching

                self.display_label.setPixmap(pix.scaled(140, 140, Qt.AspectRatioMode.KeepAspectRatio, Qt.TransformationMode.SmoothTransformation))

        else:

            # Get System Icon for other files

            file_info = QFileInfo(img_path)

            icon = QFileIconProvider().icon(file_info)

            pix = icon.pixmap(80, 80)

            self.display_label.setPixmap(pix)



        # Light Glow Shadow

        self.shadow = QGraphicsDropShadowEffect()

        self.shadow.setBlurRadius(20)

        self.shadow.setColor(QColor(0, 242, 255, 150))

        self.shadow.setOffset(0, 0)

        self.display_label.setGraphicsEffect(self.shadow)



        # Meta Box

        text_size = "12px" if is_special else "9px"

        meta_lbl = QLabel(f"{os.path.basename(img_path)}\n{datetime.fromtimestamp(os.path.getctime(img_path)).strftime('%d-%m %H:%M')}")

        meta_lbl.setStyleSheet(f"color: white; font-size: {text_size}; border: none;")

        meta_lbl.setWordWrap(True)

        meta_lbl.setAlignment(Qt.AlignmentFlag.AlignCenter)



        layout.addWidget(self.display_label)

        layout.addWidget(meta_lbl)



    def enterEvent(self, event):

        self.setStyleSheet(self.hover_style)

        self.shadow.setBlurRadius(30)



    def leaveEvent(self, event):

        self.setStyleSheet(self.normal_style)

        self.shadow.setBlurRadius(20)



    def mousePressEvent(self, event):

        if event.button() == Qt.MouseButton.LeftButton:

            self.setStyleSheet("background: #5C5CFF; border: 2px solid white;")

            if ImagePreviewOverlay:

                ImagePreviewOverlay(self.parent_page.window(), self.meta_list, self.idx)

        elif event.button() == Qt.MouseButton.RightButton:

            menu = QMenu(self)

            copy = QAction("Copy Path", self)

            copy.triggered.connect(lambda: QGuiApplication.clipboard().setText(self.img_path))

            menu.addAction(copy)

            menu.exec(event.globalPosition().toPoint())



class ResultsPage(QWidget):

    def __init__(self, parent):

        super().__init__(parent)

        self.parent = parent

        self.full_meta = []

        self.init_ui()



    def init_ui(self):

        self.main_layout = QVBoxLayout(self)

        

        # Logo Label (No white box)

        self.logo_label = QLabel("Memo Trigger")

        self.logo_label.setAlignment(Qt.AlignmentFlag.AlignCenter)

        self.logo_label.setStyleSheet("color: white; font-weight: bold;")

        self.main_layout.addWidget(self.logo_label)



        self.scroll = QScrollArea()

        self.scroll.setWidgetResizable(True)

        self.scroll.setStyleSheet("""

            QScrollArea { background: transparent; border: none; }

            QScrollBar:vertical { background: rgba(255,255,255,30); width: 12px; border-radius: 6px; }

            QScrollBar::handle:vertical { background: #00F2FF; min-height: 20px; border-radius: 6px; margin: 2px; }

        """)

        

        self.scroll_content = QWidget()

        self.boxes_layout = QVBoxLayout(self.scroll_content)

        

        self.oldest_box = self.create_section("Oldest Files")

        self.all_images_box = self.create_section("All Files", has_filters=True)

        self.latest_box = self.create_section("Latest Files")

        

        self.boxes_layout.addWidget(self.oldest_box['frame'])

        self.boxes_layout.addWidget(self.all_images_box['frame'])

        self.boxes_layout.addWidget(self.latest_box['frame'])

        

        self.scroll.setWidget(self.scroll_content)

        self.main_layout.addWidget(self.scroll)



        # Bottom Buttons

        btn_layout = QHBoxLayout()

        self.again_btn = self.create_styled_button("AGAIN")

        self.home_btn = self.create_styled_button("HOME")

        

        self.again_btn.clicked.connect(self.handle_again)

        self.home_btn.clicked.connect(self.parent.show_dashboard)

        

        btn_layout.addWidget(self.again_btn)

        btn_layout.addWidget(self.home_btn)

        self.main_layout.addLayout(btn_layout)



    def create_section(self, title, has_filters=False):

        frame = QFrame()

        frame.setStyleSheet("background: rgba(0,0,0,180); border: 2px solid white; border-radius: 20px;")

        v_layout = QVBoxLayout(frame)

        

        header_layout = QHBoxLayout()

        t_label = QLabel(title)

        t_label.setStyleSheet("color: white; font-weight: bold; font-size: 22px; border: none;")

        

        if has_filters:

            self.filter_label = t_label

            self.btn_oldest = self.create_filter_btn("Oldest to Latest", True)

            self.btn_latest = self.create_filter_btn("Latest to Oldest", False)

            

            self.btn_oldest.clicked.connect(lambda: self.resort_data(False))

            self.btn_latest.clicked.connect(lambda: self.resort_data(True))

            

            header_layout.addWidget(self.btn_oldest)

            header_layout.addStretch()

            header_layout.addWidget(t_label)

            header_layout.addStretch()

            header_layout.addWidget(self.btn_latest)

            v_layout.addLayout(header_layout)

        else:

            t_label.setAlignment(Qt.AlignmentFlag.AlignCenter)

            v_layout.addWidget(t_label)

        

        grid_widget = QWidget()

        grid_layout = QGridLayout(grid_widget)

        v_layout.addWidget(grid_widget)

        return {'frame': frame, 'grid': grid_layout}



    def create_filter_btn(self, text, is_selected):

        btn = QPushButton(text)

        btn.setFixedSize(140, 35)

        btn.setCursor(Qt.CursorShape.PointingHandCursor)

        style = "background: #5C5CFF; color: white; border: 2px solid #00F2FF; border-radius: 10px; font-weight: bold;" if is_selected else \

                "background: transparent; color: white; border: 1px solid white; border-radius: 10px;"

        btn.setStyleSheet(style)

        return btn



    def create_styled_button(self, text):

        btn = QPushButton(text)

        btn.setFixedSize(120, 45)

        btn.setCursor(Qt.CursorShape.PointingHandCursor)

        btn.setStyleSheet("QPushButton { background: white; color: #001A33; border-radius: 20px; font-weight: bold; font-size: 14px; }"

                          "QPushButton:hover { background: #00F2FF; border: 2px solid white; }")

        return btn



    def resort_data(self, reverse):

        self.btn_oldest.setStyleSheet(self.get_filter_style(not reverse))

        self.btn_latest.setStyleSheet(self.get_filter_style(reverse))

        self.full_meta.sort(key=lambda x: x[1], reverse=reverse)

        self.rebuild_grid()



    def get_filter_style(self, active):

        return "background: #5C5CFF; color: white; border: 2px solid #00F2FF; border-radius: 10px; font-weight: bold;" if active else \

               "background: transparent; color: white; border: 1px solid white; border-radius: 10px;"



    def start_loading(self):

        db_path = os.path.join(os.getcwd(), "DB.json")

        if not os.path.exists(db_path): return

        with open(db_path, "r") as f:

            paths = json.load(f).get("images", [])



        self.loader_thread = ImageLoaderThread(paths)

        self.loader_thread.image_loaded.connect(self.on_image_loaded)

        self.loader_thread.start()



    def on_image_loaded(self, data):

        if data["type"] == "meta_ready":

            self.full_meta = data["data"]

            self.filter_label.setText(f"All Files ({len(self.full_meta)})")

            self.rebuild_grid()



    def rebuild_grid(self):

        # Clear grids

        for box in [self.oldest_box, self.all_images_box, self.latest_box]:

            for i in reversed(range(box['grid'].count())):

                w = box['grid'].itemAt(i).widget()

                if w: w.setParent(None)



        # Fill grids

        for idx, item in enumerate(self.full_meta):

            row, col = divmod(idx, 5)

            self.all_images_box['grid'].addWidget(HoverIcon(item[0], self.full_meta, idx, self), row, col)



        # Special Boxes (First 3 and Last 3 from original chronological order)

        sorted_meta = sorted(self.full_meta, key=lambda x: x[1])

        for i in range(min(3, len(sorted_meta))):

            self.oldest_box['grid'].addWidget(HoverIcon(sorted_meta[i][0], sorted_meta, i, self, True), 0, i)

        

        for i in range(min(3, len(sorted_meta))):

            idx = len(sorted_meta) - 1 - i

            self.latest_box['grid'].addWidget(HoverIcon(sorted_meta[idx][0], sorted_meta, idx, self, True), 0, 2-i)



    def handle_again(self):

        if hasattr(self.parent, 'show_options'): self.parent.show_options()



    def showEvent(self, event):

        super().showEvent(event)

        self.start_loading()

        self.refresh_ui()



    def refresh_ui(self):

        f_size = min(int(self.width() * 0.07), 40)

        self.logo_label.setFont(QFont("Impact", f_size))



    def resizeEvent(self, event):

        self.refresh_ui()

        super().resizeEvent(event)
